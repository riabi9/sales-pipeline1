⚠ One-time setup needed

The client_enquiries table doesn't exist in your Supabase project yet. Run this SQL once in your Supabase SQL Editor, then refresh the page.

-- Step 1 — table for incoming client enquiries
CREATE TABLE IF NOT EXISTS client_enquiries (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    created_at timestamptz DEFAULT now(),
    ref text,
    client_name text NOT NULL,
    client_email text NOT NULL,
    client_company text,
    client_phone text,
    client_country text,
    project_ref text,
    message text,
    items jsonb,
    item_count int,
    status text DEFAULT 'new',
    source text DEFAULT 'public_catalogue'
);

-- Wipe any leftover policies from previous attempts
DROP POLICY IF EXISTS "anon_insert_enquiries"   ON client_enquiries;
DROP POLICY IF EXISTS "auth_read_enquiries"     ON client_enquiries;
DROP POLICY IF EXISTS "auth_update_enquiries"   ON client_enquiries;
DROP POLICY IF EXISTS "auth_delete_enquiries"   ON client_enquiries;
DROP POLICY IF EXISTS "anyone_can_submit"       ON client_enquiries;
DROP POLICY IF EXISTS "team_can_read"           ON client_enquiries;
DROP POLICY IF EXISTS "team_can_update"         ON client_enquiries;
DROP POLICY IF EXISTS "team_can_delete"         ON client_enquiries;

ALTER TABLE client_enquiries ENABLE ROW LEVEL SECURITY;

-- Anyone (TO public = anon + authenticated + everyone) can SUBMIT
CREATE POLICY "anyone_can_submit" ON client_enquiries
    FOR INSERT TO public WITH CHECK (true);

-- Only logged-in team members can READ / MANAGE
CREATE POLICY "team_can_read"   ON client_enquiries FOR SELECT TO authenticated USING (true);
CREATE POLICY "team_can_update" ON client_enquiries FOR UPDATE TO authenticated USING (true);
CREATE POLICY "team_can_delete" ON client_enquiries FOR DELETE TO authenticated USING (true);

-- Step 2 — make sure anonymous visitors can READ stock
DROP POLICY IF EXISTS "anon_read_stock" ON stock;
CREATE POLICY "anon_read_stock" ON stock
    FOR SELECT TO public USING (true);

DROP POLICY IF EXISTS "anon_read_stock_lists" ON stock_lists;
CREATE POLICY "anon_read_stock_lists" ON stock_lists
    FOR SELECT TO public USING (true);

-- Verification — should return 4 enquiry policies
SELECT policyname, cmd, roles FROM pg_policies
WHERE tablename = 'client_enquiries' ORDER BY policyname;
