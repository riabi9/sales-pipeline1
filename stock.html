⚠️ One-time Supabase Setup Required

The stock_lists table doesn't exist in your Supabase project yet.
Run this SQL once in your Supabase SQL Editor, then try again.
-- Step 1: Drop old table if it was partially created
DROP TABLE IF EXISTS stock_lists CASCADE;

-- Step 2: Create stock_lists with text id (matches your stock table)
CREATE TABLE stock_lists (
    id text PRIMARY KEY DEFAULT gen_random_uuid()::text,
    name text NOT NULL,
    created_at timestamptz DEFAULT now()
);
ALTER TABLE stock_lists ENABLE ROW LEVEL SECURITY;
CREATE POLICY "read_lists"   ON stock_lists FOR SELECT USING (true);
CREATE POLICY "insert_lists" ON stock_lists FOR INSERT WITH CHECK (auth.uid() IS NOT NULL);
CREATE POLICY "manage_lists" ON stock_lists FOR ALL   USING (auth.uid() IS NOT NULL);

-- Step 3: Add list_id to stock table (plain text, no foreign key)
ALTER TABLE stock ADD COLUMN IF NOT EXISTS list_id text;
CREATE INDEX IF NOT EXISTS stock_list_id_idx ON stock (list_id);

-- Step 4: Add profile fields for offer builder
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS full_name text;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS company_name text;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS company_subtitle text;

-- Step 5: Add multi-product category support
ALTER TABLE stock ADD COLUMN IF NOT EXISTS product_category text DEFAULT 'flat';
ALTER TABLE stock ADD COLUMN IF NOT EXISTS section text;
